<!DOCTYPE html PUBLIC "-//W3C//DTD XHTML 1.0 Transitional//EN" "http://www.w3.org/TR/xhtml1/DTD/xhtml1-transitional.dtd">
<html xmlns="http://www.w3.org/1999/xhtml" xml:lang="da" lang="da">
<head>
  <title>Sidens titel</title>
  <link rel="stylesheet" href="http://virk.dk/integration/thirdparty.css" type="text/css" media="screen" />
</head>
<body class="virkpartner">
  <ol class="train">
    [Indhold af train]
  </ol>
  <div id="content" class="train small">
    <form>
      <div class="fieldset">
        <fieldset>
          <legend><span>Overskrift til indberetning ("Stamdata" i eksemplet ovenfor)</span></legend>
          [...]
        </fieldset>
      </div>
    </form>
  </div>
</body>
</html>
